# Algoritmo básico de integración Monte Carlo
def monte_carlo_integration(f, a, b, n):
    suma = 0
    for i in range(n):
        x = random.uniform(a, b)
        suma += f(x)
    return (b - a) * suma / n

# Estimación de π
